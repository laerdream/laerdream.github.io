## [ 폰트 폴더 이동 ]
cd /usr/share/fonts

## [ 원격지 폰트 다운로드 ]
sudo wget 폰트파일 저장 주소

## [ 폰트 폴더 생성 ]
sudo mkdir nanum

## [ 폰트 압축해제 ]
sudo tar -xvf nanum.tar

## [ 폰트 등록 ]
sudo fc-cache -fv

## [ 폰트 등록확인 ]
fc-list

## 폰트위치
cd /usr/share/fonts
sudo mkdir googlefonts

## 폰트등록
sudo fc-cache -fv
